on_startup = {
	set_variable = {
		var = UFA_Federal_Unity
		value = 100
	}
}

on_daily_UFA = {
	effect = {
		if = {
			limit = {
				check_variable = {
					stability > UFA_Federal_Unity
				}
			}
			set_stability = federal_unity
		}
	}
}

on_daily_UFA = {
	effect = {
		if = {
			limit = {
				check_variable = {
					war_support > UFA_Federal_Unity
				}
			}
			set_war_support = federal_unity
		}
	}
}

	effect = {
		if = {
			limit = {

			}
			add_to_variable = {
				var = UFA_Federal_Unity
				value = -0.0025
			}
		}
	}
