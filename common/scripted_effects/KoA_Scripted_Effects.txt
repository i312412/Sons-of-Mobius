KOA_Switch_Opposition_Idea_Meta = {
	meta_effect = {
		text = {
			KOA = {
				if = {
					limit = {
						check_variable = {
							[CURRENT_IDEA_LEVEL_VAR] > 0
						}
					}
					if = {
						limit = {
							has_idea = [IDEA_NAME]_[OLD_LEVEL]
						}
						remove_ideas = [IDEA_NAME]_[OLD_LEVEL]
					}
				}
				if = {
					limit = {
						AND = {
							check_variable = {
								[NEW_IDEA_LEVEL_VAR] > 0
							}
							check_variable = {
								[CURRENT_IDEA_LEVEL_VAR] < 11
							}
						}
					}
					add_ideas = [IDEA_NAME]_[NEW_LEVEL]
					set_variable = {
						KOA_current_populist_idea_level = [NEW_LEVEL]
					}
					if = {
						limit = {
							has_global_flag = [HELP_FLAG]
						}
						add_stability = -0.05
						clr_global_flag = [HELP_FLAG]
					}
					clear_variable = NEW_IDEA_LEVEL_VAR
				}
			}
		}
		CURRENT_IDEA_LEVEL_VAR = "[?current_idea_level_var_name]"
		NEW_IDEA_LEVEL_VAR = "[?new_idea_level_var_name]"
		MAX_IDEA_LEVEL = "[?max_idea_level]"
		MIN_IDEA_LEVEL = "[?min_idea_level]"
		HELP_FLAG = "[?help_flag_var]"
		OLD_LEVEL = "[?[CURRENT_IDEA_LEVEL_VAR]]|.0]"
		NEW_LEVEL = "[?[NEW_IDEA_LEVEL_VAR]|.0]"
	}
}

KOA_equalize_opposition = {
	KOA = { 
		if = {
			limit = {
				has_idea = KOA_Populist_Domination
			}
			remove_ideas = KOA_Populist_Domination
		}
		if = {
			limit = {
				has_idea = KOA_Nobles_Domination
			}
			remove_ideas = KOA_Nobles_Domination
		}
	}
}

KOA_compare_opposition = {
	if = {
		limit = {
			check_variable = {
				nobles_influence < populist_influence
			}
		}
		KOA_enable_populist_domination = yes
	}
	else_if = {
		limit = {
			check_variable = {
				nobles_influence > populist_influence
			}
		}
		KOA_enable_nobles_domination = yes
	}
	else = {
		if = {
			limit = {
				check_variable = {
					var = nobles_influence
					value = populist_influence
					compare = equals
				}
			}
			KOA_equalize_opposition = yes
		}
	}
}

KOA_increase_populist_influence = {
	if = {
		limit = {
			NOT = {
				has_global_flag = KOA_Populist_vs_Noble
			}
		}
	}
	else_if = {
		limit = {
			OR = {
				check_variable = { populist_influence < 100 }
				check_variable = { nobles_influence > 0 }
			}
		}
		add_to_variable = { populist_influence = 10 }
		subtract_from_variable = { nobles_influence = 10 }
	}
	else = {
		set_variable = { populist_influence = 100 }
		set_variable = { nobles_influence = 0 }
	}
	hidden_effect = {
		KOA_compare_opposition = yes
	}
	KOA_check_nobles_levity = yes
}
