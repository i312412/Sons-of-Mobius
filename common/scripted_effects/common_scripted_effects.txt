Enable_Trade_Embargo = {
    ROOT = {
        add_opinion_modifier = {
            target = PREV
            modifier = Trade_Embargo
        }
    }
    PREV = {
        add_opinion_modifier = {
            target = ROOT
            modifier = Trade_Embargo_Opinion
        }
    }
}

Disable_Trade_Embargo = {
    ROOT = {
        remove_opinion_modifier = {
            target = PREV
            modifier = Trade_Embargo
        }
    }
    PREV = {
        remove_opinion_modifier = {
            target = ROOT
            modifier = Trade_Embargo_Opinion
        }
    }
}
